SELECT c.name,
       COUNT(o.id) AS order_count,
       SUM(o.total) AS revenue
FROM customers c
JOIN orders o ON o.customer_id = c.id
WHERE o.created_at >= date_trunc('month', NOW())
GROUP BY c.id, c.name
HAVING COUNT(o.id) > 3
ORDER BY order_count DESC
