1-- Films sortis en 2006 par durée de location moyenne
2SELECT
3  f.title,
4  f.rating,
5  AVG(EXTRACT(DAY FROM (r.return_date - r.rental_date))) AS avg_rental_days,
6  COUNT(r.rental_id) AS total_rentals
7FROM film f
8JOIN inventory i ON i.film_id = f.film_id
9JOIN rental r ON r.inventory_id = i.inventory_id
10WHERE f.release_year = 2006
11  AND r.return_date IS NOT NULL
12GROUP BY f.film_id, f.title, f.rating
13ORDER BY avg_rental_days DESC
14LIMIT 20;
